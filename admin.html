async function uploadParaImgBB(input) { const file = input.files[0]; if(!file) return; const status = document.getElementById('status-upload'); const preview = document.getElementById('img-preview'); const btnFinal = document.getElementById('btn-final'); status.style.display = 'block'; status.innerText = "SUBINDO FOTO..."; preview.style.display = 'none'; btnFinal.disabled = true; const formData = new FormData(); formData.append('image', file); // CHAVE DE API (Verifique se é esta mesma na sua conta ImgBB) const apiKey = '666e632b7e6157143615951666e92c22'; try { const response = await fetch(`https://api.imgbb.com/1/upload?key=${apiKey}`, { method: 'POST', body: formData }); const data = await response.json(); if (data.success) { const urlFinal = data.data.url; document.getElementById('prod-img').value = urlFinal; preview.src = urlFinal; preview.style.display = 'block'; status.style.display = 'none'; btnFinal.disabled = false; btnFinal.innerText = "CONCLUIR LANÇAMENTO"; console.log("Upload feito com sucesso: ", urlFinal); } else { // Se o ImgBB responder com erro (Ex: chave inválida) throw new Error(data.error.message); } } catch (error) { console.error("Erro detalhado do ImgBB:", error); alert("ERRO NO UPLOAD: " + error.message); status.innerText = "Erro ao subir. Tente novamente."; btnFinal.disabled = false; btnFinal.innerText = "TENTAR NOVAMENTE"; } }
